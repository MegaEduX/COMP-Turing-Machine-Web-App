/*	
 *	Checks if the input is a binary palindrome.
 *	Taken and adapted from http://morphett.info/turing/turing.html
 */

[s0] {
	0 _ r 1o
	1 _ r 1i
	_ _ * acc
}

[1o] {
	_ _ l 2o
	* * r 1o
}

[1i] {
	_ _ l 2i
	* * r 1i
}

[2o] {
	0 _ l 3a
	_ _ * acc
	* * * rej
}

[2i] {
	1 _ l 3a
	_ _ * acc
	* * * rej
}

[3a] {
	_ _ * acc
	* * l 4a
}

[4a] {
	_ _ r s0
	* * l 4a
}

[acc] {
	* : r acc2
}

[acc2] {
	* ) * accept
}

[rej] {
	_ : r rej2
	* _ l rej
}

[rej2] {
	* ( * reject
}
